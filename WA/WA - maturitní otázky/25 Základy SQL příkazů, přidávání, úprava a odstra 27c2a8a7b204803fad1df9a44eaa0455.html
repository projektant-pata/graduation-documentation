Insert

  • přidává nové záznamy do tabulky
  • pokud není uveden výčet sloupců. nutnost zadaní všech sloupců (i když tam hodnota je defaultně, třeba index, či null) v přesném pořadí

Příklad INSERT bez výčtu sloupců

-- Nutnost všech sloupců (včetně ID) (ID, jmeno, vek, mesto_ID)
INSERT INTO studenti 
VALUES (67, 'Tomas Buzik', 19, 4);

Příklad INSERT s výčtem sloupců

-- u ID se použije AUTO-INCREMENT
INSERT INTO studenti (jmeno, vek, mesto_ID) 
VALUES ('Tomas Buzik', 19, 4);

Příklad více INSERT 

-- u ID se použije AUTO-INCREMENT
INSERT INTO studenti (jmeno, vek, mesto_ID) 
VALUES ('Tomáš Buzik', 19, 4),
		('Šmoulinka', 18, 4),
		('Martin Bombač', 18, 4);

Příklad INSERT se SELECT

-- INSERT studenta, který má pevně nastavený jméno a věk
-- mesto_ID se SELECTne, pokud nějaký PSC začíná ma 53 (v našem příkladě Pardubice)
INSERT INTO studenti (jmeno, vek, mesto_ID)
SELECT 'Tomas Buzik', 19, mesto_ID
FROM mesta
WHERE PSC LIKE '530%';

Příklad chytáku na syntaxi obdobnou UPDATE

-- standartní insert nemá tuto formu, ale MySQL ji podporuje
-- není nutno zadávat ID, má AUTO-INCREMENT
INSERT INTO studenti 
SET jmeno = 'Tomas Buzik', vek = 19, mesto_ID = 4;

Update

  • upravuje existující záznamy v databázi

Příklad UPDATE

-- nastaví na tyto hodnoty všechny záznamy => nechceme
UPDATE studenti SET jmeno = 'Tomas Cernej Pan', vek = 18, mesto_ID = 4;

Příklad UPDATE se SELECT

-- nastaví na tyto hodnoty všechny záznamy => nechceme
UPDATE studenti 
SET jmeno = 'Tomas Cernej Pan', vek = 18, mesto_ID = 4 
WHERE ID = 67;;

Příklad UPDATE se SELECT s agregačními akcemi

-- má narozeniny 🥳
UPDATE studenti 
SET vek = vek + 1 
WHERE ID = 67;

Příklad UPDATE s JOIN

-- všichni v PCE mají narozeniny 🎉
UPDATE studenti
JOIN mesta ON studenti.mesto_ID = mesta.mesto_ID
SET studenti.vek = studenti.vek + 1
WHERE mesta.PSC = '530 00';

Delete

  • maže řádky z tabulky
  • může se využít s WHERE
  • NIKDY NEPOUŽÍVAT *

Příklad DELETE celé tabulky

DELETE FROM studenti; --!!BEZ *!!

Příklad DELETE s WHERE

DELETE FROM studenti WHERE vek < 18;

Příklad DELETE s JOIN

-- smaže všechny Pardubičáky
DELETE studenti
FROM studenti
JOIN mesta ON studenti.mesto_ID = mesta.mesto_ID
WHERE mesta.PSC = '530 00';

Truncate

  • nespouští triggery (typ reakce na smazání dat)
  • rychlejší jak DELETE bez where
  • resetuje AUTO-INCREMENT

Příklad TRUNCATE tabulky

TRUNCATE TABLE studenti;

Drop

  • může smazat
    • tabulku včetně záznamů
    • celou databázi

Příklad DROP tabulky

DROP TABLE studenti;

Příklad DROP databáze

DROP DATABASE MaturitniOtazkyWP;
